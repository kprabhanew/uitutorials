JS Module Patterns (just a few)

1. Classic Pattern

Example 1 

    var sampleModule = (function() {
    //private variables

    var a = 10;
    var b = 20;

    //private functions
    var getSum = function() {
        return (a + b);
    }

    var getProduct = function() {
        return (a * b);
    }

    //expose public object/members
    return {
        sum: getSum,
        product: getProduct
    }
})();

2. CommonJS Pattern

Example 2 

  var sampleModule = function() {
    //privatet variables

    var a = 10;
    var b = 20;

    //public functions
    this.sum = function() {
        return (a + b);
    };

    this.product = function() {
        return (a * b);
    };
};

//export this class
module.exports = sampleModule;

//Importing into other module
//1. require statement will fetch the module and execute the defenition and assign into 'mod'
//2. This 'mod' will contain entier defenition then create the instance and access the members
var mod = require('sampleModule');
var o = new mod();
console.log(o.sum());
console.log(o.product());

3. AMD Pattern

Example 2 

  define([], function() {
    //privat variables
    var a = 10;
    var b = 20;

    //expose public object/members

    return {
        sum: function() {
            return (a + b);
        },
        product: function() {
            return (a + b);
        }
    };
})
 
//Using this pattern we can directly use the instance, no need to create new instance. 
//Because you are sending instance you are not sending functions anymore
var o = require('sampleModule');
console.log(o.sum());
console.log(o.product());

//Sampe with different syntax, it means it is using callback method
require(['sampleModule'], function(o) {
    console.log(o.sum());
    console.log(o.product());
})

4. TypeScript Language / Pattern

  • Objects in a module can be exposed to public
  • Using "export"
  • Objects (of a module) can be obtained (and used) in other modules
  • Using "import"
Example 

    class SampleModule {

    //private variables
    private a: number = 10;
    private b: number = 20;

    //expose public members
    public sum() {
        return (this.a + this.b);
    };

    public product() {
        return (this.a * this.b);
    };
};

//Needs to use 'export' keyword to export this class, to use in some other module
export { SampleModule };

import { SampleModule } from "./SampleModule";

let o = new SampleModule();
console.log(o.sum());
console.log(o.product());
