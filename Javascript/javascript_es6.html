Destructuring Object

Example : app.js 

    //Destructuring Object
let person = {
    name: "Ryan",
    age: 30,
    location: "Toranto"
};

person.age; //30
person["age"]; //30

//let { age: personAge } = person;
//let { age, location: currentLocation } = person;
let key = "age";
let {
    [key]: keyAge
} = person;

console.log(keyAge);
