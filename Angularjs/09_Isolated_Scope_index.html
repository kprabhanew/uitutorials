09 Isoclated Scope using '&' symbol

& - function binding (call external function from directive)

Example : index.html 

   <!DOCTYPE html>
<html ng-app="app">

<head>
    <meta charset="utf-8">
    <title>Angular Intro</title>
    <link rel="stylesheet" type="text/css" href="https://maxcdn.bootstrapcdn.com/bootstrap/3.3.7/css/bootstrap.min.css">
    <script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/3.1.1/jquery.min.js"></script>
    <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.3.15/angular.min.js"></script>
    <script src="app.js"></script>
</head>

<body ng-app="app">

    <div id="div1" ng-controller="sample">
        Controller: a = {{a}}, b = {{b}}
        <br> a: <input ng-model="a" />
        <br>b: <input ng-model="b" />
        <br><button ng-click="doSum(a, b)">Sum</button>
        <hr>
        <!-- first way (with first way in template-->
        Directive1 Instance 1:
        <!--doSum : defenition of parent Not execution only sending or assigned to ext-sum -->
        <div message ext-sum="doSum"></div><br>

        <!-- second way (with second way in template-->
        <!-- sending as an object -->
        <!-- Directive1 Instance 2: <div message ext-sum="doSum(m,n)"></div><br> -->

        <!-- third way (with third way in js-->
        <!-- Directive1 Instance 1: <div message just-sum="doSum"></div><br> -->
    </div>
</body>

</html>
Example : app.js 

    var app = angular.module('app', []);

app.controller('sample', ['$scope', function($scope) {

    $scope.a = 10;
    $scope.b = 20;

    //Sum Calculate Parent Conroller 
    $scope.doSum = function(x, y) {
        alert("Sum = " + (parseInt(x) + parseInt(y)));
    }

}]);

app.directive('message', function() {

    return {
        templateUrl: 'info-msg.htm',
        //& - function binding (call external function from directive)
        scope: {
            extSum: '&'

            //third way (with third way in markup) with  different specification of attribute
            //extSum: '&justSum'
        },

        //Directive based controller
        controller: function($scope, $element, $attrs) {

            //Add a button execute the function
            $scope.doProcess = function() {
                //Execute External Sum
                //$scope - Local Scope / Isolate Scope
                //$scope.p,q - From Current Scope
                $scope.extSum({ m: $scope.p, n: $scope.q });
            };

            //Access Parent directly from isolated scope
            // $scope.doProcess2 = function() {
            //     //$scope.extSum()($scope.p, $scope.q);            
            //     //$scope.extSum()($scope.p, $scope.q * $scope.$parent.a);
            //     $scope.extSum({ m: $scope.p, n: ($scope.q * $scope.$parent.a) })
            // }

        }


    }

});
Example : info-msg.html 

    <div>
    <!--Local p,q as part of isolated scope for directive-->
    p = {{p}}, q = {{q}}
    <br>p: <input ng-model="p" />
    <br>q: <input ng-model="q" />
    <br>
    <!-- first way (with first way in template-->
    <button ng-click="extSum()(p,q)">Sum</button>

    <!-- sceond way (with sceond way in template-->
    <!--<button ng-click="extSum({m:p,n:q})">Sum</button>-->

    <button ng-click="doProcess()">Sum</button>

</div>
